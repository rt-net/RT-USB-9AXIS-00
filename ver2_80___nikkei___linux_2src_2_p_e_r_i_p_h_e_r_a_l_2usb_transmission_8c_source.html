70  if( ((LPC_GPIO0->DATA & 0x8)>>3))
71  {
72  USB_Connect(1);
73  while (!USB_Configuration){
74  wait1msec(1);
75  count ++;
76  if(count > 6000){
77  flag_usb_config = 0;
78  return 0;
79  };
80  };
81  flag_usb_config = 1;
82  return 1;
83  }
84  else

90  }
