336 void SystemCoreClockUpdate (void) /* Get Core Clock Frequency */
337 {
338  uint32_t wdt_osc = 0;
339 
340  /* Determine clock frequency according to clock register values */
341  switch ((LPC_SYSCON->WDTOSCCTRL >> 5) & 0x0F) {
342  case 0: wdt_osc = 400000; break;
343  case 1: wdt_osc = 500000; break;
344  case 2: wdt_osc = 800000; break;
345  case 3: wdt_osc = 1100000; break;
346  case 4: wdt_osc = 1400000; break;
347  case 5: wdt_osc = 1600000; break;
348  case 6: wdt_osc = 1800000; break;
349  case 7: wdt_osc = 2000000; break;
350  case 8: wdt_osc = 2200000; break;
351  case 9: wdt_osc = 2400000; break;
352  case 10: wdt_osc = 2600000; break;
353  case 11: wdt_osc = 2700000; break;
354  case 12: wdt_osc = 2900000; break;
355  case 13: wdt_osc = 3100000; break;
356  case 14: wdt_osc = 3200000; break;
357  case 15: wdt_osc = 3400000; break;
358  }
359  wdt_osc /= ((LPC_SYSCON->WDTOSCCTRL & 0x1F) << 1) + 2;
360 
361  switch (LPC_SYSCON->MAINCLKSEL & 0x03) {
362  case 0: /* Internal RC oscillator */
363  SystemCoreClock = __IRC_OSC_CLK;
364  break;
365  case 1: /* Input Clock to System PLL */
366  switch (LPC_SYSCON->SYSPLLCLKSEL & 0x03) {
367  case 0: /* Internal RC oscillator */
368  SystemCoreClock = __IRC_OSC_CLK;
369  break;
370  case 1: /* System oscillator */
371  SystemCoreClock = __SYS_OSC_CLK;
372  break;
373  case 2: /* WDT Oscillator */
374  SystemCoreClock = wdt_osc;
375  break;
376  case 3: /* Reserved */
377  SystemCoreClock = 0;
378  break;
379  }
380  break;
381  case 2: /* WDT Oscillator */
382  SystemCoreClock = wdt_osc;
383  break;
384  case 3: /* System PLL Clock Out */
385  switch (LPC_SYSCON->SYSPLLCLKSEL & 0x03) {
386  case 0: /* Internal RC oscillator */
387  if (LPC_SYSCON->SYSPLLCTRL & 0x180) {
388  SystemCoreClock = __IRC_OSC_CLK;
389  } else {
390  SystemCoreClock = __IRC_OSC_CLK * ((LPC_SYSCON->SYSPLLCTRL & 0x01F) + 1);
391  }
392  break;
393  case 1: /* System oscillator */
394  if (LPC_SYSCON->SYSPLLCTRL & 0x180) {
395  SystemCoreClock = __SYS_OSC_CLK;
396  } else {
397  SystemCoreClock = __SYS_OSC_CLK * ((LPC_SYSCON->SYSPLLCTRL & 0x01F) + 1);
398  }
399  break;
400  case 2: /* WDT Oscillator */
401  if (LPC_SYSCON->SYSPLLCTRL & 0x180) {
402  SystemCoreClock = wdt_osc;
403  } else {

427 void SystemInit (void)

431 #if (SYSOSC_SETUP) /* System Oscillator Setup */

434  LPC_SYSCON->PDRUNCFG &= ~(1 << 5); /* Power-up System Osc */

437  LPC_SYSCON->SYSPLLCLKSEL = SYSPLLCLKSEL_Val; /* Select PLL Input */

442 #if (SYSPLL_SETUP) /* System PLL Setup */

444  LPC_SYSCON->PDRUNCFG &= ~(1 << 7); /* Power-up SYSPLL */
445  while (!(LPC_SYSCON->SYSPLLSTAT & 0x01)); /* Wait Until PLL Locked */
446 #endif
447 #endif
448 #if (WDTOSC_SETUP) /* Watchdog Oscillator Setup*/

453  LPC_SYSCON->MAINCLKUEN = 0x01; /* Update MCLK Clock Source */
454  LPC_SYSCON->MAINCLKUEN = 0x00; /* Toggle Update Register */
